class Solution {
    public int[] solution(int n, int s) {
        if(n > s){
            int[] error = {-1};
            return error;
        }
        int[] answer = new int[n];
        int a = s / n, b = s % n;
        for(int i = 0; i < n; i++){
            answer[i] = a;
        }
        for(int i = n - 1; i > n - 1 - b; i--){
            answer[i] += 1;
        }
        return answer;
    }
}
