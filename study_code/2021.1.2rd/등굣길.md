class Solution {
    public int Mod = 1000000007;
    public int solution(int m, int n, int[][] puddles) {
        int answer = 0;
        int[][] dp = new int[n][m];
        dp[0][0] = 1;
        for(int[] element : puddles){
            dp[element[1] - 1][element[0] - 1] = -1;
        }
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
                if(dp[i][j] == -1){
                    dp[i][j] = 0;
                    continue;
                }
                if(i != 0){
                    dp[i][j] += dp[i - 1][j] % Mod;
                }
                if(j != 0){
                    dp[i][j] += dp[i][j - 1] % Mod;
                }
            }
        }
        return dp[n - 1][m - 1] % Mod;
    }
}

class Solution {
    public int Mod = 1000000007;
    public int[] X = {1, 0};
    public int[] Y = {0, 1};
    public int answer, M, N;
    public boolean[][] visited;
    public void dfs(int x, int y){
        if(x == N - 1 && y == M - 1){
            answer++;
            return;
        }
        for(int i = 0; i < 2; i++){
            int nextX = x + X[i];
            int nextY = y + Y[i];
            if(nextX < 0 || nextX >= N){
                continue;
            }
            if(nextY < 0 || nextY >= M){
                continue;
            }
            if(visited[nextX][nextY]){
                continue;
            }
            visited[nextX][nextY] = true;
            dfs(nextX, nextY);
            visited[nextX][nextY] = false;
        }
        return;
    }
    public int solution(int m, int n, int[][] puddles) {
        answer = 0;
        M = m;
        N = n;
        visited = new boolean[n][m];
        for(int[] element : puddles){
            visited[element[1] - 1][element[0] - 1] = true;
        }
        dfs(0, 0);
        return answer;
    }
}
