import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.StringTokenizer;
import java.util.Arrays;
import java.util.LinkedList;
import java.util.Queue;

class Main {
    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        int N = Integer.parseInt(st.nextToken());
        int K = Integer.parseInt(st.nextToken());
        int M = Integer.parseInt(st.nextToken());
        ArrayList<ArrayList<Integer>> graph = new ArrayList<>();
        for (int i = 0; i <= N + M; i++) {
            graph.add(new ArrayList<Integer>());
        }
        int busstop = N + 1;
        for (int i = 0; i < M; i++) {
            st = new StringTokenizer(br.readLine(), " ");
            for (int j = 0; j < K; j++) {
                int num = Integer.parseInt(st.nextToken());
                graph.get(busstop).add(num);
                graph.get(num).add(busstop);
            }
            busstop++;
        }
        int[] distance = new int[N + M + 1];
        boolean[] visited = new boolean[N + M + 1];
        Arrays.fill(distance, Integer.MAX_VALUE);
        Queue<Integer> queue = new LinkedList<Integer>();
        queue.offer(1);
        distance[1] = 1;
        visited[1] = true;
        while (!queue.isEmpty()) {
            int node = queue.poll();
            for (Integer j : graph.get(node)) {
                if (visited[j]) {
                    continue;
                }
                if (j > N) {
                    distance[j] = Math.min(distance[node] + 1, distance[j]);
                } else {
                    distance[j] = Math.min(distance[node], distance[j]);
                }
                visited[j] = true;
                queue.offer(j);
            }
        }
        if (distance[N] == Integer.MAX_VALUE) {
            System.out.println(-1);
        } else {
            System.out.println(distance[N]);
        }
        br.close();
    }
}

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.StringTokenizer;
import java.util.Arrays;
import java.util.LinkedList;
import java.util.Queue;

class Main {
    public static ArrayList<ArrayList<Integer>> graph;

    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        int N = Integer.parseInt(st.nextToken());
        int K = Integer.parseInt(st.nextToken());
        int M = Integer.parseInt(st.nextToken());
        graph = new ArrayList<>();
        for (int i = 0; i <= N; i++) {
            graph.add(new ArrayList<Integer>());
        }
        int[] arr;
        for (int i = 0; i < M; i++) {
            st = new StringTokenizer(br.readLine(), " ");
            arr = new int[K];
            for (int j = 0; j < K; j++) {
                arr[j] = Integer.parseInt(st.nextToken());
            }
            for (int j = 0; j < K; j++) {
                for (int k = 0; k < K; k++) {
                    if (j == k)
                        continue;
                    graph.get(arr[j]).add(arr[k]);
                }
            }
        }
        int[] distance = new int[N + 1];
        boolean[] visited = new boolean[N + 1];
        Arrays.fill(distance, Integer.MAX_VALUE);
        Queue<Integer> queue = new LinkedList<Integer>();
        queue.offer(1);
        distance[1] = 1;
        while (!queue.isEmpty()) {
            int node = queue.poll();
            visited[node] = true;
            for (Integer j : graph.get(node)) {
                if (visited[j]) {
                    continue;
                }
                if (distance[j] == Integer.MAX_VALUE || distance[node] + 1 <= distance[j]) {
                    distance[j] = distance[node] + 1;
                    queue.add(j);
                } else {
                    continue;
                }
            }
        }
        System.out.println(distance[N]);
        br.close();
    }
}
