import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.HashSet;
import java.util.HashMap;

class Main {
    public static HashMap<Character, Integer> map;
    public static ArrayList<Character> WordArr;
    public static boolean[] visited;
    public static int answer, Arr_size, N;
    public static char[][] input;

    public static int mapCalculate() {
        int answer = 0;
        for (int i = 0; i < N; i++) {
            int tmp = 0;
            for (Character word : input[i]) {
                tmp *= 10;
                tmp += map.get(word);
            }
            answer += tmp;
        }
        return answer;
    }

    public static void permutation(int count) {
        if (count == Arr_size) {
            int num = mapCalculate();
            if (num > answer)
                answer = num;
            return;
        }
        for (int i = 0; i < 10; i++) {
            if (!visited[i]) {
                visited[i] = true;
                map.put(WordArr.get(count), i);
                permutation(count + 1);
                visited[i] = false;
            }
        }
    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());
        map = new HashMap<Character, Integer>();
        HashSet<Character> set = new HashSet<Character>();
        input = new char[N][];
        for (int i = 0; i < N; i++) {
            char[] line = br.readLine().toCharArray();
            input[i] = line;
            for (char word : line) {
                set.add(word);
            }
        }
        WordArr = new ArrayList<>(set);
        Arr_size = WordArr.size();
        visited = new boolean[10];
        answer = 0;
        permutation(0);
        System.out.println(answer);
        br.close();
    }
}
