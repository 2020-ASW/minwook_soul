    private static int[][] arr;
    private static boolean[][] visited;

    public static void dfs(int x, int y, boolean isDigging, int count, int before) {
        visited[x][y] = true;
        for (int i = 0; i < 4; i++) {
            int nextX = x + X[i];
            int nextY = y + Y[i];
            if (nextX < 0 || nextX >= N) {
                continue;

                continue;
            }

                continue;
            }

                    continue;

                dfs(nextX, nextY, isDigging, count + 1, arr[x][y]);
            }
            continue;
        }
        return;
    }
