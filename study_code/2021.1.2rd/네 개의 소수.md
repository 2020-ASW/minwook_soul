import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.ArrayList;

class Main {
    public static boolean isDecimal(int N) {
        int Max = (int) Math.sqrt(N);
        for (int i = 2; i <= Max; i++) {
            if (N % i == 0) {
                return false;
            }
        }
        return true;
    }

    public static void Count(int N) {
        ArrayList<Integer> arr = new ArrayList<>();
        for (int i = 2; i < N; i++) {
            if (isDecimal(i)) {
                arr.add(i);
            }
        }
        int Size = arr.size();
        int a, b, c, d, sum;
        for (a = 0; a < Size; a++) {
            for (b = a; b < Size; b++) {
                sum = arr.get(a) + arr.get(b);
                if (sum >= N) {
                    continue;
                }
                for (c = b; c < Size; c++) {
                    sum = arr.get(a) + arr.get(b) + arr.get(c);
                    if (sum >= N) {
                        continue;
                    }
                    for (d = c; d < Size; d++) {
                        sum = arr.get(a) + arr.get(b) + arr.get(c) + arr.get(d);
                        if (sum != N) {
                            continue;
                        } else {
                            System.out.println(arr.get(a) + " " + arr.get(b) + " " + arr.get(c) + " " + arr.get(d));
                            return;
                        }
                    }
                }
            }
        }
        System.out.println(-1);
        return;
    }

    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int N = Integer.parseInt(br.readLine());
        Count(N);
        br.close();
    }
}

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.ArrayList;

class Main {
    public static boolean isDecimal(int N) {
        int Max = (int) Math.sqrt(N);
        for (int i = 2; i <= Max; i++) {
            if (N % i == 0) {
                return false;
            }
        }
        return true;
    }

    public static void Count(int N) {
        ArrayList<Integer> arr = new ArrayList<>();
        for (int i = 2; i < N; i++) {

                arr.add(i);
            }

        int Size = arr.size();
        if (N % 2 == 0) {
            if (N < 8) {
                System.out.println(-1);
                return;

            if (N < 9) {
                System.out.println(-1);
                return;

                if (N != arr.get(i) + arr.get(j)) {
                    continue;
                } else {
                    System.out.println(arr.get(i) + " " + arr.get(j));

                }

    public static void main(String[] args) throws Exception {
