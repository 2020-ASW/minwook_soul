import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

class Node {
    int x;
    int y;
    int count;
    int k;

    Node(int x, int y, int k, int count) {
        this.x = x;
        this.y = y;
        this.k = k;
        this.count = count;
    }
}

class Main {
    private static int[] X = { 1, 0, -1, 0 };
    private static int[] Y = { 0, 1, 0, -1 };
    private static int[] HorseX = { 1, 2, 2, 1, -1, -2, -2, -1 };
    private static int[] HorseY = { -2, -1, 1, 2, -2, -1, 1, 2 };
    private static int[][] arr;
    private static boolean[][][] visited;

    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int K = Integer.parseInt(br.readLine());
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        int W = Integer.parseInt(st.nextToken());
        int H = Integer.parseInt(st.nextToken());
        arr = new int[H][W];
        visited = new boolean[H][W][K + 1];
        for (int i = 0; i < H; i++) {
            st = new StringTokenizer(br.readLine(), " ");
            for (int j = 0; j < W; j++) {
                arr[i][j] = Integer.parseInt(st.nextToken());
            }
        }
        int nextX, nextY;
        Queue<Node> queue = new LinkedList<Node>();
        queue.add(new Node(0, 0, K, 0));
        while (!queue.isEmpty()) {
            Node node = queue.poll();
            int x = node.x;
            int y = node.y;
            int k = node.k;
            int count = node.count;
            if (x == H - 1 && y == W - 1) {
                System.out.println(count);
                return;
            }
            if (x < 0 || x >= H) {
                continue;
            }
            if (y < 0 || y >= W) {
                continue;
            }
            if (arr[x][y] == 1) {
                continue;
            }
            if (visited[x][y][k]) {
                continue;
            }
            visited[x][y][k] = true;
            for (int i = 0; i < 4; i++) {
                nextX = x + X[i];
                nextY = y + Y[i];
                queue.add(new Node(nextX, nextY, k, count + 1));
            }
            if (k <= 0) {
                continue;
            }
            for (int i = 0; i < 8; i++) {
                nextX = x + HorseX[i];
                nextY = y + HorseY[i];
                queue.add(new Node(nextX, nextY, k - 1, count + 1));
            }
        }
        System.out.println("-1");
        br.close();
    }
}

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

class Main {
    private static int answer, W, H;
    private static int[] X = { 1, 0, -1, 0 };
    private static int[] Y = { 0, 1, 0, -1 };
    private static int[] HorseX = { 1, 2, 2, 1, -1, -2, -2, -1 };
    private static int[] HorseY = { -2, -1, 1, 2, -2, -1, 1, 2 };
    private static int[][] arr;
    private static boolean[][] visited;

    public static void dfs(int x, int y, int k, int count) {
        if (x == H - 1 && y == W - 1) {
            if (count < answer)
                answer = count;
            return;
        }
        if (count > arr[x][y]) {
            return;
        }
        arr[x][y] = count;
        int nextX, nextY;
        for (int i = 0; i < 4; i++) {
            nextX = x + X[i];
            nextY = y + Y[i];
            if (nextX < 0 || nextX >= H) {
                continue;
            }
            if (nextY < 0 || nextY >= W) {
                continue;
            }
            if (visited[nextX][nextY]) {
                continue;
            }
            visited[x][y] = true;
            dfs(nextX, nextY, k, count + 1);
            visited[x][y] = false;
        }
        for (int i = 0; i < 6; i++) {
            nextX = x + HorseX[i];
            nextY = y + HorseY[i];
            if (nextX < 0 || nextX >= H) {
                continue;
            }
            if (nextY < 0 || nextY >= W) {
                continue;
            }
            if (visited[nextX][nextY]) {
                continue;
            }
            if (k <= 0) {
                continue;
            }
            visited[x][y] = true;
            dfs(nextX, nextY, k - 1, count + 1);
            visited[x][y] = false;
        }
    }

    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int K = Integer.parseInt(br.readLine());

                    arr[i][j] = Integer.MAX_VALUE;
                if (arr[i][j] == 1)

            }
        }
        answer = Integer.MAX_VALUE;
